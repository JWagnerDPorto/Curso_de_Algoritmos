N, S: Inteiro
resp: caractere
inicio

     Repita

              S <- S + N
              Escreva("Você quer continuar? [S/N] ")
              Leia(resp)
     Ate (resp = "N")
     Escreval("A soma de todos os valores digitados e ", S)
